```bash
#!/usr/bin/env bash
# scripts/circleci/post_devrev_comment.sh
# Usage: bash scripts/circleci/post_devrev_comment.sh "<ISSUE_ID>" "<Comment body (markdown)>"
# Called by the agent via Bash tool to post mid-run comments to DevRev.
set -euo pipefail

ISSUE_ID="$1"
BODY="$2"

if [ -z "${ISSUE_ID}" ] || [ -z "${BODY}" ]; then
  echo "Usage: post_devrev_comment.sh <ISSUE_ID> <BODY>" >&2
  exit 1
fi

curl -sf -X POST "https://api.devrev.ai/timeline-entries.create" \
  -H "Authorization: ${DEVREV_PAT}" \
  -H "Content-Type: application/json" \
  -d "{
    \"type\": \"timeline_comment\",
    \"object\": \"${ISSUE_ID}\",
    \"body\": $(echo "${BODY}" | jq -Rs .),
    \"visibility\": \"external\"
  }"
```

**devrev-web/scripts/circleci/run_nitpick_fix_agent.sh** (rebuilt for two-stage flow per D-11 through D-44):
```bash
#!/usr/bin/env bash
# scripts/circleci/run_nitpick_fix_agent.sh
# Two-stage nitpick fix agent runner.
# Env vars: NITPICK_ISSUE_ID (required), NITPICK_MODE (analysis|fix|revision, default: analysis), DEVREV_PAT (required)
set -euo pipefail

# --- Configuration ---
ISSUE_ID="${NITPICK_ISSUE_ID:?'NITPICK_ISSUE_ID must be set'}"
MODE="${NITPICK_MODE:-analysis}"
PROMPT_FILE="scripts/circleci/nitpick_fix_prompt.md"
MAX_TURNS=65
DEVREV_PAT="${DEVREV_PAT:?'DEVREV_PAT must be set'}"

echo "[nitpick] Starting ${MODE} run for ${ISSUE_ID}"

# --- Install Claude Code CLI ---
npm install -g @anthropic-ai/claude-code

# --- Bedrock auth (from CircleCI aws-dev-account context) ---
export CLAUDE_CODE_USE_BEDROCK=1
export AWS_REGION="${AWS_REGION:-us-east-1}"
export ANTHROPIC_DEFAULT_SONNET_MODEL='us.anthropic.claude-sonnet-4-6'

# --- Fetch issue metadata from DevRev ---
ISSUE_DATA=$(curl -sf "https://api.devrev.ai/works.get?id=${ISSUE_ID}" \
  -H "Authorization: ${DEVREV_PAT}")
TITLE=$(echo "${ISSUE_DATA}" | jq -r '.work.title // ""')
DESCRIPTION=$(echo "${ISSUE_DATA}" | jq -r '.work.body // ""')
WORK_ID=$(echo "${ISSUE_DATA}" | jq -r '.work.id // ""')

# --- Fetch ALL timeline entries for context injection (D-36) ---
TIMELINE=$(curl -sf "https://api.devrev.ai/timeline-entries.list?object=${WORK_ID}" \
  -H "Authorization: ${DEVREV_PAT}" \
  | jq '[.timeline_entries[]? | {body, author: .created_by.display_name, date: .created_date}]')

# --- Update nitpick_stage (D-31) ---
case "${MODE}" in
  analysis) STAGE="Analyzing" ;;
  fix)      STAGE="Fixing" ;;
  revision) STAGE="Revising" ;;
  *)        STAGE="Analyzing" ;;
esac
curl -sf -X POST "https://api.devrev.ai/works.update" \
  -H "Authorization: ${DEVREV_PAT}" \
  -H "Content-Type: application/json" \
  -d "{\"id\": \"${WORK_ID}\", \"custom_fields\": {\"nitpick_stage\": \"${STAGE}\"}}" || true

# --- Prepare workspace (D-16, D-37) ---
# Branch naming: fix/ISS-XXXX-short-description
SHORT_DESC=$(echo "${TITLE}" | tr '[:upper:]' '[:lower:]' | sed 's/[^a-z0-9]/-/g' | head -c 30 | sed 's/-$//')
BRANCH_NAME="fix/${ISSUE_ID}-${SHORT_DESC}"

if [ "${MODE}" = "revision" ] || [ "${MODE}" = "fix" ]; then
  # D-37: For fix/revision, try checkout existing branch first
  if git ls-remote --heads origin "${BRANCH_NAME}" 2>/dev/null | grep -q "${BRANCH_NAME}"; then
    git fetch origin "${BRANCH_NAME}"
    git checkout "${BRANCH_NAME}"
    git pull origin "${BRANCH_NAME}" --rebase
    echo "[nitpick] Checked out existing branch: ${BRANCH_NAME}"
  else
    git checkout -b "${BRANCH_NAME}"
    echo "[nitpick] Created new branch: ${BRANCH_NAME}"
  fi
else
  # Analysis mode: stay on main (no branch needed, no PR created)
  echo "[nitpick] Analysis mode: staying on main branch"
fi

# --- Concurrency guard (D-44) ---
if [ "${MODE}" = "fix" ]; then
  EXISTING_PR=$(gh pr list --head "${BRANCH_NAME}" --state open --json number -q '.[0].number' 2>/dev/null || echo "")
  if [ -n "${EXISTING_PR}" ]; then
    echo "[nitpick] PR #${EXISTING_PR} already open for ${ISSUE_ID}. Skipping fix run."
    exit 0
  fi
fi

# --- Inject context into prompt template (D-39) ---
export ISSUE_ID TITLE DESCRIPTION MODE TIMELINE WORK_ID BRANCH_NAME
PROMPT=$(envsubst '${ISSUE_ID} ${TITLE} ${DESCRIPTION} ${MODE} ${TIMELINE} ${WORK_ID} ${BRANCH_NAME}' < "${PROMPT_FILE}")

# --- Run agent ---
echo "[nitpick] Invoking claude -p in ${MODE} mode (max ${MAX_TURNS} turns)"
claude -p "${PROMPT}" \
  --model sonnet \
  --max-turns "${MAX_TURNS}" \
  --permission-mode bypassPermissions \
  --allowedTools "Read" "Grep" "Glob" "Edit" "Write" \
    "Bash(pnpm run workspace-lint *)" "Bash(pnpm run affected:typecheck *)" \
    "Bash(git add *)" "Bash(git commit *)" "Bash(git push *)" \
    "Bash(git diff *)" "Bash(git log *)" "Bash(git status *)" \
    "Bash(git checkout *)" "Bash(git branch *)" \
    "Bash(gh pr create *)" "Bash(gh pr view *)" "Bash(gh pr edit *)" \
    "Bash(nx affected *)" \
    "Bash(bash scripts/circleci/post_devrev_comment.sh *)" \
  --output-format json \
  --verbose \
  2>&1 | tee /tmp/agent-output.json

AGENT_EXIT=$?

# --- Post-run stage update (D-31) ---
case "${MODE}" in
  analysis)
    # After analysis, agent should have posted fix options. Set stage to Awaiting choice.
    curl -sf -X POST "https://api.devrev.ai/works.update" \
      -H "Authorization: ${DEVREV_PAT}" \
      -H "Content-Type: application/json" \
      -d "{\"id\": \"${WORK_ID}\", \"custom_fields\": {\"nitpick_stage\": \"Awaiting choice\"}}" || true
    ;;
  fix|revision)
    # After fix/revision, agent should have created/updated PR. Set stage to In review.
    curl -sf -X POST "https://api.devrev.ai/works.update" \
      -H "Authorization: ${DEVREV_PAT}" \
      -H "Content-Type: application/json" \
      -d "{\"id\": \"${WORK_ID}\", \"custom_fields\": {\"nitpick_stage\": \"In review\"}}" || true
    ;;
esac

# --- Post-run: Link PR to DevRev issue (D-34) ---
if [ "${MODE}" = "fix" ] || [ "${MODE}" = "revision" ]; then
  PR_NUMBER=$(gh pr list --head "${BRANCH_NAME}" --state open --json number -q '.[0].number' 2>/dev/null || echo "")
  if [ -n "${PR_NUMBER}" ]; then
    PR_URL="https://github.com/$(gh repo view --json nameWithOwner -q .nameWithOwner)/pull/${PR_NUMBER}"
    # Post PR link as comment (D-34: both link API and comment for visibility)
    bash scripts/circleci/post_devrev_comment.sh "${ISSUE_ID}" \
      "PR created: [#${PR_NUMBER}](${PR_URL})" || true
  fi
fi

# --- Post-run guardrails ---
# Branch naming validation
if [ "${MODE}" = "fix" ] || [ "${MODE}" = "revision" ]; then
  CURRENT_BRANCH=$(git rev-parse --abbrev-ref HEAD)
  if [[ ! "${CURRENT_BRANCH}" =~ ^fix/ISS-[A-Z0-9]+-.*$ ]] && [[ ! "${CURRENT_BRANCH}" =~ ^fix/${ISSUE_ID}-.*$ ]]; then
    echo "[nitpick] WARNING: Branch '${CURRENT_BRANCH}' does not match expected pattern"
  fi
fi

# --- Error reporting (D-27, D-28) ---
if [ "${AGENT_EXIT}" -ne 0 ]; then
  CIRCLE_BUILD_URL="${CIRCLE_BUILD_URL:-https://app.circleci.com}"
  bash scripts/circleci/post_devrev_comment.sh "${ISSUE_ID}" \
    "The fix attempt ran into trouble. Check the [CI logs](${CIRCLE_BUILD_URL}) for details. Let me know if you'd like me to try again." || true
fi

echo "[nitpick] ${MODE} run complete for ${ISSUE_ID} (exit: ${AGENT_EXIT})"
exit ${AGENT_EXIT}
```

**Structure:**

```markdown
# Nitpick Fix Agent -- ${ISSUE_ID}

## Your Role
You are a senior frontend engineer working in the devrev-web monorepo. You follow the repo's CLAUDE.md conventions exactly. You communicate findings and fixes conversationally on the DevRev issue timeline.

**Current mode: ${MODE}**

## Issue Context
- **DevRev Issue:** ${ISSUE_ID}
- **Title:** ${TITLE}
- **Bug Description and Metadata:**
${DESCRIPTION}

## Conversation History
The following is the full timeline of comments on this issue (your previous analysis, user replies, etc.):
${TIMELINE}

---

## MODE: analysis
**When ${MODE} = "analysis"**

### Objective
Analyze the bug and present fix options. Do NOT create a branch or PR.

### Flow (turns 1-60)
1. **Understand the bug** (turns 1-5):
   - Parse the issue metadata: symptom, page URL, DOM path, CSS properties, component classes
   - Cross-check the captured element against the title (detect wrong-element captures)
   - Map URL to feature area using devrev-web path patterns

2. **Find the source** (turns 6-20):
   - Use progressive narrowing: Glob -> Grep -> Read
   - Find the React component chain responsible for the affected element
   - Trace theme/styling: useTheme -> theme config -> CSS variable -> component
   - Identify whether this is a DS bug (shared component) vs local bug (single consumer)

3. **Formulate fix options** (turns 21-40):
   - Determine 2-4 distinct approaches to fix the bug
   - For each option: identify affected file(s), write before/after code snippets
   - Rank by: minimal change, convention adherence, side-effect risk

4. **Post analysis to DevRev** (turns 41-60):
   - Use `bash scripts/circleci/post_devrev_comment.sh "${ISSUE_ID}" "<comment>"` to post
   - Format the comment as:
     ```
     **Root cause:** [one clear line explaining why this happens]

     Here are the fix options:

     **Option 1: [approach name]**
     File: `path/to/file.tsx`
     ```tsx
     // Before
     [current code]

     // After
     [proposed change]
     ```
     Approach: [plain language explanation]

     **Option 2: [approach name]**
     ...

     Which approach works for you? (Or describe what you'd prefer and I'll adapt.)
     ```
   - After posting, you are DONE. Do not create a branch or PR.

---

## MODE: fix
**When ${MODE} = "fix"**

### Objective
Implement the user's chosen fix direction and create a draft PR.

### Flow (turns 1-60)
1. **Read the user's choice** (turns 1-3):
   - Look at the MOST RECENT user comment in the conversation history
   - Understand their direction -- this is natural language, not just "option 2"
   - They may say "option 2 but use className instead" -- that's a composite instruction

2. **Implement the fix** (turns 4-30):
   - You are on branch: ${BRANCH_NAME}
   - Make the minimal change that implements the user's chosen direction
   - Use design system tokens only (no inline styles, no magic values)
   - Run verification: `nx affected --target=typecheck` and `pnpm run workspace-lint`
   - Commit with `PRE_COMMIT_AUTO_RUN=1` and format: `fix(<scope>): <description>`

3. **Self-review** (turns 31-45):
   - Run `git diff main...HEAD` to see all changes
   - Ask yourself: Does this actually fix the reported bug? Is the scope minimal? Are DS tokens used correctly?
   - If issues found: fix, verify, re-commit
   - Maximum 3 self-review rounds

4. **Create draft PR** (turns 46-55):
   - `git push -u origin ${BRANCH_NAME}`
   - `gh pr create --draft --title "fix(<scope>): <description>" --body "<PR template>"`

   - PR body must include: Description, Changes (In Scope / Not In Scope), DevRev work item link
   - Include `DevRev Work Item: ${ISSUE_ID}` in PR body

5. **Post results to DevRev** (turns 56-60):
   - Get the PR number from `gh pr view --json number -q .number`
   - Post comment via helper script:
     ```

     Take a look and let me know if anything needs adjusting.
     ```

6. **Tag codeowners** (turn 60):

   - Post comment: "Tagging the code owners for review: @owner1 @owner2 (owners of [paths]). PR #<NUMBER> is ready for their review."
     ```
